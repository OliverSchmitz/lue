\documentclass[crop, tikz]{standalone}

\definecolor{l colour}{HTML}{5e81ac}
\definecolor{u colour}{HTML}{a3be8c}
\definecolor{e colour}{HTML}{bf616a}

\tikzset{
    letter/.style={
        draw=black,
        line width=10pt,
    },
    plot/.style={
        draw=black,
        line width=10pt,
    },
}

% NOTE: Whatever you do, keep the favicon square. Tweaking the spacing helps with that.

\begin{document}
    \begin{tikzpicture}[cap=round]
        \pgfmathsetmacro \rangex {{1.5}}
        \pgfmathsetmacro \minx {{-\rangex}}
        \pgfmathsetmacro \nrsamples {{100}}
        \pgfmathsetmacro \spacing {{0.375}}

        \pgfmathsetmacro \letteridx {{0}}
        \pgfmathsetmacro \offset {{0}}
        \draw[
            letter,
            domain=\minx : 0.00,
            samples=\nrsamples,
            ] plot[smooth] ({\offset + \x}, {\x*\x}) [l colour];

        \pgfmathsetmacro \letteridx {{1}}
        \pgfmathsetmacro \offset {{\letteridx * \spacing}}
        \draw[
            letter,
            domain=\minx : 0.00,
            samples=\nrsamples,
            ] plot[smooth] ({\offset + \x}, {\x*\x}) -- ({\offset}, {1.5 * 1.5}) [u colour];

        \pgfmathsetmacro \letteridx {{2}}
        \pgfmathsetmacro \offset {{\letteridx * \spacing}}
        \draw[
            letter,
            domain=\minx : 0.00,
            samples=\nrsamples,
            ] ({\offset}, {\minx * \minx}) -- plot[smooth] ({\offset + \x}, {\x*\x}) [e colour];
        \draw[
            letter,
        ] ({\offset + 0.75 * \minx}, {(0.75 * \minx) * (0.75 * \minx)}) --
        ({\offset}, {(0.75 * \minx) * (0.75 * \minx)}) [e colour];
    \end{tikzpicture}
\end{document}
