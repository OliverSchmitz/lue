\documentclass[crop, tikz]{standalone}

\tikzset{
    letter/.style={
        draw=black,
        line width=2.75pt,
    },
    plot/.style={
        draw=black,
        line width=2.75pt,
    },
}

\begin{document}
    \begin{tikzpicture}[cap=round]
        \pgfmathsetmacro \rangex {{1.5}}
        \pgfmathsetmacro \minx {{-\rangex}}
        \pgfmathsetmacro \spacing {{5}}
        \pgfmathsetmacro \nrsamples {{100}}
        \pgfmathsetmacro \spacing {{0.2}}

        \pgfmathsetmacro \letteridx {{0}}
        \pgfmathsetmacro \offset {{0}}
        \draw[
            letter,
            domain=\minx : 0.00,
            samples=\nrsamples,
        ] plot[smooth] ({\offset + \x}, {\x*\x});

        \pgfmathsetmacro \letteridx {{1}}
        \pgfmathsetmacro \offset {{\letteridx * (\rangex + \spacing)}}
        \draw[
            letter,
            domain=\minx : 0.00,
            samples=\nrsamples,
        ] plot[smooth] ({\offset + \x}, {\x*\x}) -- ({\offset}, {1.5 * 1.5});

        \pgfmathsetmacro \letteridx {{2}}
        \pgfmathsetmacro \offset {{\letteridx * (\rangex + \spacing)}}
        \draw[
            letter,
            domain=\minx : 0.00,
            samples=\nrsamples,
        ] ({\offset}, {\minx * \minx}) -- plot[smooth] ({\offset + \x}, {\x*\x});
        \draw[
            letter,
        ] ({\offset + 0.75 * \minx}, {(0.75 * \minx) * (0.75 * \minx)}) --
            ({\offset}, {(0.75 * \minx) * (0.75 * \minx)});
    \end{tikzpicture}
\end{document}
